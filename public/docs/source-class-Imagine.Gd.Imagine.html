 44:      * {@inheritdoc}

 46:     public function create(BoxInterface $size, ColorInterface $color = null)
 47:     {
 48:         $width  = $size->getWidth();
 49:         $height = $size->getHeight();
 50: 
 51:         $resource = imagecreatetruecolor($width, $height);
 52: 
 53:         if (false === $resource) {
 54:             throw new RuntimeException('Create operation failed');
 55:         }
 56: 
 57:         $palette = null !== $color ? $color->getPalette() : new RGB();
 58:         $color = $color ? $color : $palette->color('fff');
 59: 
 60:         if (!$color instanceof RGBColor) {
 61:             throw new InvalidArgumentException('GD driver only supports RGB colors');
 62:         }
 63: 
 64:         $index = imagecolorallocatealpha($resource, $color->getRed(), $color->getGreen(), $color->getBlue(), round(127 * (100 - $color->getAlpha()) / 100));
 65: 
 66:         if (false === $index) {
 67:             throw new RuntimeException('Unable to allocate color');
 68:         }
 69: 
 70:         if (false === imagefill($resource, 0, 0, $index)) {
 71:             throw new RuntimeException('Could not set background color fill');
 72:         }
 73: 
 74:         if ($color->getAlpha() >= 95) {
 75:             imagecolortransparent($resource, $index);
 76:         }
 77: 
 78:         return $this->wrap($resource, $palette, new MetadataBag());
 79:     }

140:     private function wrap($resource, PaletteInterface $palette, MetadataBag $metadata)
141:     {
142:         if (!imageistruecolor($resource)) {
143:             list($width, $height) = array(imagesx($resource), imagesy($resource));
144: 
145:             // create transparent truecolor canvas
146:             $truecolor   = imagecreatetruecolor($width, $height);
147:             $transparent = imagecolorallocatealpha($truecolor, 255, 255, 255, 127);
148: 
149:             imagefill($truecolor, 0, 0, $transparent);
150:             imagecolortransparent($truecolor, $transparent);
151: 
152:             imagecopymerge($truecolor, $resource, 0, 0, 0, 0, $width, $height, 100);
153: 
154:             imagedestroy($resource);
155:             $resource = $truecolor;
156:         }
157: 
158:         if (false === imagealphablending($resource, false) || false === imagesavealpha($resource, true)) {
159:             throw new RuntimeException('Could not set alphablending, savealpha and antialias values');
160:         }
161: 
162:         if (function_exists('imageantialias')) {
163:             imageantialias($resource, true);
164:         }
165: 
166:         return new Image($resource, $palette, $metadata);
167:     }
