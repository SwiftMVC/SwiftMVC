330:     /**
331:      * @param array  $options
332:      * @param string $path
333:      */
334:     private function prepareOutput(array $options, $path = null)
335:     {
336:         if (isset($options['format'])) {
337:             $this->imagick->setImageFormat($options['format']);
338:         }
339: 
340:         if (isset($options['animated']) && true === $options['animated']) {
341:             $format = isset($options['format']) ? $options['format'] : 'gif';
342:             $delay = isset($options['animated.delay']) ? $options['animated.delay'] : null;
343:             $loops = isset($options['animated.loops']) ? $options['animated.loops'] : 0;
344: 
345:             $options['flatten'] = false;
346: 
347:             $this->layers->animate($format, $delay, $loops);
348:         } else {
349:             $this->layers->merge();
350:         }
351:         $this->applyImageOptions($this->imagick, $options, $path);
352: 
353:         // flatten only if image has multiple layers
354:         if ((!isset($options['flatten']) || $options['flatten'] === true) && count($this->layers) > 1) {
355:             $this->flatten();
356:         }
357:     }

407:     {

520:     /**
521:      * Returns a color given a pixel, depending the Palette context
522:      *
523:      * Note : this method is public for PHP 5.3 compatibility
524:      *
525:      * @param \ImagickPixel $pixel
526:      *
527:      * @return ColorInterface
528:      *
529:      * @throws InvalidArgumentException In case a unknown color is requested
530:      */
531:     public function pixelToColor(\ImagickPixel $pixel)
532:     {
533:         static $colorMapping = array(
534:             ColorInterface::COLOR_RED     => \Imagick::COLOR_RED,
535:             ColorInterface::COLOR_GREEN   => \Imagick::COLOR_GREEN,
536:             ColorInterface::COLOR_BLUE    => \Imagick::COLOR_BLUE,
537:             ColorInterface::COLOR_CYAN    => \Imagick::COLOR_CYAN,
538:             ColorInterface::COLOR_MAGENTA => \Imagick::COLOR_MAGENTA,
539:             ColorInterface::COLOR_YELLOW  => \Imagick::COLOR_YELLOW,
540:             ColorInterface::COLOR_KEYLINE => \Imagick::COLOR_BLACK,
541:             // There is no gray component in \Imagick, let's use one of the RGB comp
542:             ColorInterface::COLOR_GRAY    => \Imagick::COLOR_RED,
543:         );
544: 
545:         $alpha = $this->palette->supportsAlpha() ? (int) round($pixel->getColorValue(\Imagick::COLOR_ALPHA) * 100) : null;
546:         $palette = $this->palette();
547: 
548:         return $this->palette->color(array_map(function ($color) use ($palette, $pixel, $colorMapping) {
549:             if (!isset($colorMapping[$color])) {
550:                 throw new InvalidArgumentException(sprintf('Color %s is not mapped in Imagick', $color));
551:             }
552:             $multiplier = 255;
553:             if ($palette->name() === PaletteInterface::PALETTE_CMYK) {
554:                 $multiplier = 100;
555:             }
556: 
557:             return $pixel->getColorValue($colorMapping[$color]) * $multiplier;
558:         }, $this->palette->pixelDefinition()), $alpha);
559:     }

573:     {
574:         if (!isset(static::$colorspaceMapping[$palette->name()])) {
575:             throw new InvalidArgumentException(sprintf('The palette %s is not supported by Imagick driver', $palette->name()));
576:         }

583:             throw new RuntimeException('Your version of Imagick does not support colorspace conversions.');

595:             }

597:             $this->profile($palette->profile());

599:         } catch (\ImagickException $e) {

649:     /**
650:      * Internal
651:      *
652:      * Applies options before save or output
653:      *
654:      * @param \Imagick $image
655:      * @param array    $options
656:      * @param string   $path
657:      *
658:      * @throws InvalidArgumentException
659:      * @throws RuntimeException
660:      */
661:     private function applyImageOptions(\Imagick $image, array $options, $path)
662:     {
663:         if (isset($options['format'])) {
664:             $format = $options['format'];
665:         } elseif ('' !== $extension = pathinfo($path, \PATHINFO_EXTENSION)) {
666:             $format = $extension;
667:         } else {
668:             $format = pathinfo($image->getImageFilename(), \PATHINFO_EXTENSION);
669:         }
670: 
671:         $format = strtolower($format);
672: 
673:         $options = $this->updateSaveOptions($options);
674: 
675:         if (isset($options['jpeg_quality']) && in_array($format, array('jpeg', 'jpg', 'pjpeg'))) {
676:             $image->setImageCompressionQuality($options['jpeg_quality']);
677:         }
678: 
679:         if ((isset($options['png_compression_level']) || isset($options['png_compression_filter'])) && $format === 'png') {
680:             // first digit: compression level (default: 7)
681:             if (isset($options['png_compression_level'])) {
682:                 if ($options['png_compression_level'] < 0 || $options['png_compression_level'] > 9) {
